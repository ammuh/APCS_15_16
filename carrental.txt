public class CarRental{
    private String make;
    private String model;
    private String plate;
    private String code;
    public static void main(String args[]){
        CarRental car = new CarRental("Tesla", "Model x", "CPR 607");
        car.computeCode();
        System.out.print(car.getCode());
        
    }
    public CarRental( String mk, String mdl, String lis){
        make = mk;
        model = mdl;
        plate = lis;
    }
    public void computeCode(){
        String prefix = this.plate.subString(0,2);
        String suffix = this.plate.subString(4,6);
        int sum = (int)prefix.charAt(0) + (int)prefix.charAt(1) + (int)prefix.charAt(2);
        sum += (int)suffix.charAt(0) + (int)suffix.charAt(1) + (int)suffix.charAt(2) - 48*3;
        char l1 =  (char)(65 + (sum % 26));
        this.code = String(l1) + String(sum);
    }
    public String getCode(){
        return this.code;
    }
}
